114 void parent_selection(POPULATION *population, int num_elitism) {
115 
116  /*
117  * Selection can be performed via:
118  * fitness selection (Roulette Wheel)
119  * Elitism (most fit chromosomes pass onto next generation automatically)
120  */
121 
122  // holds the sum of each chromosome probability to be selected
123  float cumulative_prob[population->num_chromosomes]; // create local array to not allocate memory for it
124  calculate_cumulative_probability(population, cumulative_prob);
125 
126  /*
127  * if any chromosome is to be passed via elitism
128  * we start to iterate from next chromosome to it
129  */
130

132  POPULATION temp_population = deep_copy_population(population);
133 
134  // choose chromosomes

136  for (size_t i = num_elitism; i < population->num_chromosomes; ++i) {
137  temp_chromo = population->chromosomes + i;

143 void calculate_cumulative_probability(POPULATION *population, float *cumulative_prob_arr) {
144  /*
145  * Probability is calculated via the formula
146  * Pi = Ai / ∑Aj
147  * Pi = probability of each chromosome
148  * Ai = fitness of each chromosome
149  * ∑Aj = sum of all chromosomes fitness
150  */
151 
152  // ∑Aj
153  float sum_chromosome_fitness = sum_population_fitness(population);
154 
155  /*
156  * We can think about the sum_prob as a roulette
157  * Each section of the roulette represent the chance (percentage)
158  * of being chosen
159  */
160  float roulette[population->num_chromosomes];
161 
162  for (size_t i = 0; i < population->num_chromosomes; ++i) {
163  CHROMOSOME *temp_chromo = population->chromosomes + i;
164  // calculate prob of each chromosome
165  roulette[i] = temp_chromo->fitness_value / sum_chromosome_fitness;
166  // calculate cumulative probability of each chromosomes
167  if (i < 1) {
168  cumulative_prob_arr[i] = roulette[i];
169  } else {
170  cumulative_prob_arr[i] = roulette[i] + cumulative_prob_arr[i - 1];
171  }
172  }
173 }

184 CHROMOSOME fitness_selection(POPULATION population, const float *cumulative_prob) {
185  // range acts like the spinning mechanism of roulette
186  float range = float_in_range(0.0f, 1.0f);
187 
188  // iterate over population to find chromosome selected
189  for (size_t i = 1; i < population.num_chromosomes; ++i) {
190  CHROMOSOME *temp_chromo = population.chromosomes + i;
191  if (range <= cumulative_prob[i] && range > cumulative_prob[i - 1]) {
192  return *temp_chromo;
193  }
194  }
195 
196  return *population.chromosomes; // return first chromosome if failed
197 }

204 void breed(POPULATION *population, int num_elitism) {
205  /*
206  * Cross-over is done by combining n parent chromosomes
207  * i.e.
208  * CHROMOSOME n0 * CHROMOSOME n1 = CHROMOSOME (n0 * n1)
209  * CHROMOSOME n1 * CHROMOSOME n2 = CHROMOSOME (n1 * n2)
210  * ...
211  * CHROMOSOME nx * CHROMOSOME n0 = CHROMOSOME (nx * n0)
212  */
213 
214  // save first chromo for successful cross-over
215  CHROMOSOME saved_chromo = copy_chromosome(*population->chromosomes);
216 
217  CHROMOSOME *parent_one = NULL;
218  CHROMOSOME *parent_two = NULL;
219  CHROMOSOME *temp_chromo;
220  // select parent one and parent two
221  for (size_t i = num_elitism; i < population->num_chromosomes; ++i) {
222  temp_chromo = population->chromosomes + i;
223 
224  parent_one = temp_chromo;
225  if (i == population->num_chromosomes - 1) {
226  // set parent_two address to saved_chromo address otherwise address of parent two == parent_one
227  parent_two = &saved_chromo;
228  } else {
229  parent_two = temp_chromo + 1;
230  }
231 
232  *temp_chromo = cross_over(parent_one, parent_two);
233  }
234 
235 }

261 CHROMOSOME cross_over(CHROMOSOME *parent_one, CHROMOSOME *parent_two) {
262  // Copy parent_one
263  CHROMOSOME *child = parent_one;
264 
265  // Set initial fitness value
266  child->fitness_value = 0;
267 
268  // Shuffle genes to choose random indices
269  shuffle_genes(child);
270 
271  /*
272  * Set number of genes to half
273  * This equates to parent one genes that were selected
274  * to stay in chromosome
275  */
276  child->num_genes = parent_one->num_genes / 2;
277 
278  // Fill the rest of chromosome with parent_two genes
279  int flag = 0;
280  for (size_t i = 0; i < parent_two->num_genes; ++i) {
281  GENE *temp_parent = parent_two->genes + i;
282  for (size_t j = 0; j < child->num_genes; ++j) {
283  GENE *temp_child = child->genes + j;
284  if (temp_parent->id == temp_child->id) {
285  flag = 1;
286  break;
287  }
288  }
289  if (flag != 1) {
290  GENE *temp_child = child->genes + child->num_genes;
291  *temp_child = *temp_parent;
292  child->num_genes += 1;
293  }
294  if (child->num_genes == parent_two->num_genes) {
295  break;
296  }
297  flag = 0;
298  }
299  return *child;
300 }

302 POPULATION mutation(POPULATION population, float mutation_prob) {
303  // create random index to be selected
304  int random_gene_index = 0;
305 
306  /*
307  * For each gene of each chromosome calculate a random value
308  * to compare to the mutation probability value
309  * If mutation prob is bigger than random value
310  * mutate chromosome by swapping its genes
311  */
312  for (size_t i = 0; i < population.num_chromosomes; ++i) {
313  CHROMOSOME *temp_chromo = population.chromosomes + i;
314  for (size_t j = 0; j < temp_chromo->num_genes; ++j) {
315  GENE *temp_gene = temp_chromo->genes + j;
316  float p_range = float_in_range(0.0f, 1.0f);
317  if (p_range < mutation_prob) {
318  do {
319  // random gene index must be different than current gene index
320  random_gene_index = int_in_range(0, temp_chromo->num_genes - 1);
321  } while (random_gene_index == j);
322  swap_gene(temp_gene, temp_chromo->genes + random_gene_index);
323  }
324  }
325 
326  // calculates fitness value of chromosome
327  temp_chromo->fitness_value = calculate_fitness(temp_chromo);
328  }
329  return population;
330 }
