266 CHROMOSOME* cross_over (CHROMOSOME *parent_one, CHROMOSOME *parent_two) {
267 
268  // choose random non-repeating numbers from parent one
269  int num_of_random_numbers = parent_one->num_of_genes / 2;
270  CHROMOSOME *child = parent_one; // copies parent_one
271  child->fitness_value = 0; // makes sure that fitness value is 0
272  shuffle_genes(child, child->num_of_genes); // shuffles genes
273  child->num_of_genes = num_of_random_numbers;
274 
275  // fill the rest of child_cromo with parent_two genes
276  int flag = 0;
277  for (size_t i = 0; i < parent_two->num_of_genes ; ++i) {
278  GENE *temp_parent = parent_two->genes + i;
279  for (size_t j = 0; j < child->num_of_genes; ++j) {
280  GENE *temp_child = child->genes + j;
281  if (temp_parent->id == temp_child->id) {
282  flag = 1;
283  break;
284  }
285  }
286  if (flag != 1) {
287  GENE *temp_child = child->genes + child->num_of_genes;
288  *temp_child = *temp_parent;
289  child->num_of_genes += 1;
290  }
291  if (child->num_of_genes == parent_two->num_of_genes) break;
292  flag = 0;
293  }
294  return child;
