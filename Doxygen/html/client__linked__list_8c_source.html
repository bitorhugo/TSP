55  if (list->list_size < 1) {

57  list->head->client = create_client(name, VAT, address, phone_number, birth_day, birth_month, birth_year);
58  list->list_size = 1;
59  }
60  else {
61  CLIENT_NODE *new_node = allocate_memory_node_client();
62  new_node->client = create_client(name, VAT, address, phone_number, birth_day, birth_month, birth_year);
63 
64  if (at_head) {
65  new_node->next_node = list->head;
66  list->head = new_node;
67  list->list_size += 1;
68  }
69  else {
70  CLIENT_NODE *temp_node = list->head;
71  while (temp_node->next_node != NULL) {
72  temp_node = temp_node->next_node;
73  }
74  temp_node->next_node = new_node;
75  list->list_size += 1;
76  }
77  }

871 void split_list_in_two(CLIENT_NODE * source, CLIENT_NODE ** frontRef, CLIENT_NODE ** backRef)
872 {
873  // if the length is less than 2, handle it separately
874  if (source == NULL || source->next_node == NULL)
875  {
876  *frontRef = source;
877  *backRef = NULL;
878  return;
879  }
880 
881  CLIENT_NODE * slow = source;
882  CLIENT_NODE * fast = source->next_node;
883 
884  // advance `fast` two nodes, and advance `slow` one node
885  while (fast != NULL)
886  {
887  fast = fast->next_node;
888  if (fast != NULL)
889  {
890  slow = slow->next_node;
891  fast = fast->next_node;
892  }
893  }
894 
895  // `slow` is before the midpoint in the list, so split it in two
896  // at that point.
897  *frontRef = source;
898  *backRef = slow->next_node;
899  slow->next_node = NULL;
900 }
901 CLIENT_NODE* sorted_merge(CLIENT_NODE *a, CLIENT_NODE *b)
902 {
903  // base cases
904  if (a == NULL) {
905  return b;
906  }
907 
908  else if (b == NULL) {
909  return a;
910  }
911 
912  CLIENT_NODE *result = NULL;
913 
914  // pick either `a` or `b`, and recur
915  if (a->client.VAT <= b->client.VAT)
916  {
917  result = a;
918  result->next_node = sorted_merge(a->next_node, b);
919  }
920  else {
921  result = b;
922  result->next_node = sorted_merge(a, b->next_node);
923  }
924 
925  return result;
926 }
